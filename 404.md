<script>

  }
  if (url.includes('/Monsters')) {
    url = url.replace('/Monsters', '/monsters');

  }

  }

    url = url.replace('/Recipes.html', '/recipes.html');
    redirect = true;
  }

    window.location.href(url);
    document.head.innerHTML += `<link rel="canonical" href="${url}" />`;
  }
